## 默认展示

<Preview>
  <Story name="默认展示">

## 两张预览

<Preview>
  <Story name="两张预览">
    <PhotoProvider>
      <ImageList>
        {photoImages.map((item, index) => (
          <PhotoConsumer key={index} src={item}>
            {index < 2 ? <ViewBox viewImage={item} /> : undefined}
          </PhotoConsumer>
        ))}
      </ImageList>
    </PhotoProvider>
  </Story>
</Preview>
